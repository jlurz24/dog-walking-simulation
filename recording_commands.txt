#/bin/bash
# Capture the video
glc-capture -o figure8.glc /opt/ros/fuerte/stacks/simulator_gazebo/gazebo/gazebo/bin/gzclient -g /opt/ros/fuerte/stacks/simulator_gazebo/gazebo/lib/libgazebo_ros_paths_plugin.so
# Encode it
glc-play figure8.glc -y 1 -o - | mencoder -demuxer y4m - -ovc lavc -lavcopts vcodec=mjpeg -o figure8.avi
# Compress it
mencoder -ovc xvid -oac mp3lame -xvidencopts bitrate=878 -o figure8_c.avi figure8.avi
